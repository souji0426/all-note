  集合$A$に対して$\mathcal{I}\subseteq \mathcal{P}(A)$が集合$A$上の\textbf{ideal}$\ \defarr\ $以下の3条件を満たすもののこと.
  \begin{enumerate}
  \item $\mathcal{I}\neq \emptyset $.
  \item $\forall X,Y\in \mathcal{I}(\  X\cup Y\in \mathcal{I}\ )$.
  \item $\forall X,Y(\ X\subseteq Y \land Y\in \mathcal{I}\ \rightarrow \ X\in \mathcal{I}\ )$.
  \end{enumerate}
  集合$A$に対して$\mathcal{F}\subseteq \mathcal{P}(A)$が集合$A$上の\textbf{filter}$\ \defarr\ $以下の3条件を満たすもののこと.
  \begin{enumerate}
    \item $\mathcal{F}\neq \emptyset$.
    \item $\forall X,Y\in \mathcal{F}(\  X\cap Y\in \mathcal{F}\ )$.
    \item $\forall X,Y(\ X\subseteq Y \land X\in \mathcal{F}\ \rightarrow \ Y\in \mathcal{F}\ )$. \qedhere
  \end{enumerate}

\begin{itembox}[l]{Example \ref{example:簡単なideal, filterの例}\ \ \ \ \pageref{example:簡単なideal, filterの例}ページ}
  集合$A$に対して,
  \begin{itemize}
    \item[(1)] $\mathcal{P}(A)$は$A$上のidealかつfilter.
    \item[(2)] $\{ \emptyset \}$は$A$上のideal, $\{ A \}$は$A$上のfilter.
    \item[(3)]
      $x\in A$に対して, $\{ X\subseteq A | x\notin X  \}$は$A$上のideal,
      $\{ X\subseteq A | x\in X  \}$は$A$上のfilter.
      このようなidealを\textbf{principal ideal},
      \textbf{principal filter}と呼んだりする.
    \item[(4)]
      $S\subseteq A$に対して, $\{ X\subseteq A | X\subseteq S  \}$は$A$上のideal,
      $\{ X\subseteq A | S\subseteq X  \}$は$A$上のfilter.
  \end{itemize}

  以降$A$は無限集合として
  \begin{itemize}
    \item[(5)]
      $A$の有限部分集合全体$[A]^{<\omega}$は$A$上のideal,
      $A$の補有限集合全体$\{ X\subseteq A\ |\ |A\setminus X|<\omega\}$は$A$上のfilter.
      $|A\setminus X|<\omega$とは$A\setminus X\in [A]^{<\omega}$とも表現できる.
      またこのようなfilterを\textbf{Fr\'{e}chet filter}とも呼ぶ.

    \item[(6)]
      上の例を拡張して,
      基数$\kappa$に対して$[A]^{<\kappa}$は$A$上のideal. \qedhere
  \end{itemize}

\begin{itembox}[l]{Definition }
  $A$上のfilter\ $\mathcal{F}$が\textbf{ultra}であるとは,

\begin{itembox}[l]{Proposition \ref{proposition:ultra filterの性質まとめ}\ \ \ \ \pageref{proposition:ultra filterの性質まとめ}ページ}

      $A$上のproper filter\ $\mathcal{F}$がultraでない\ $\Longleftrightarrow$

    \item[・] $\bigcup_{i\in[n]}X_i\in\mathcal{U}\ \rightarrow\ \exists i\in[n]( X_i\in\mathcal{U} )$. \qedhere
